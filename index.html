// api/index.mjs (minimal wrapper example) import express from "express"; import { shell, etzSvg, malchutOrb, yesodOrb, binahStub, announcementsHtml, simpleHomeBody } from "../etzchaim_core.mjs"; const app = express(); app.get("/", (req, res) => res.send(shell("Home", simpleHomeBody()))); app.get("/etzchaim", (req, res) => res.send(shell("Etz Chaim", etzSvg()))); app.get("/malchut", (req, res) => res.send(shell("Malchut", malchutOrb()))); app.get("/yesod", (req, res) => res.send(shell("Yesod", yesodOrb()))); app.get("/binah", (req, res) => res.send(shell("Binah", binahStub()))); app.get("/announcements", (req, res) => res.send(shell("Announcements", announcementsHtml())) ); export default function handler(req, res) { return app(req, res); }
